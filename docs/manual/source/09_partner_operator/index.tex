        \subsection{Просмотр заказа}

                \item email покупателя \index{email покупателя};
                \item телефон покупателя \index{телефон покупателя};
                \item общую сумму заказа \index{сумма заказа};
                \item статус заказа \index{статус заказа};
                \item дату добавления заказа \index{дата добавления заказа};
                \item дату изменения заказа \index{дата изменения заказа};
                \item высылается ли по заказу сертификат \index{сертификат заказа};
                \item дату истечения бронирования\index{дата истечения бронирования};

            \subsubsection{История}
                Во вкладке <<История заказа>> доступен журнал изменения статуса заказа 
                \index{статус заказа} и обмена с \gloss{UNF} \index{1С УНФ} в виде
                таблицы со следующими столбцами (см. рис.\ref{fig:admin_func_order_history})
                \begin{enumerate}
                    \item дата события;
                    \item автор (инициатор события);
                    \item тип события;
                    \item статус (если не указан, значит не изменился с прошлого события) \index{статус заказа};
                    \item предыдущий статус (отладочная информация)
                    \item выпадающий список статусов и кнопка изменения статуса
                \end{enumerate}

                
                \begin{figure}
                    \center
                    \includegraphics[width=170mm]{05_admin_funcs/figures/05.eps}
                    \caption{Просмотр заказа. История.}
                    \label{fig:admin_func_order_history}
                \end{figure}

                \paragraph{Исполнение заказа}
                    \index{исполнение заказа}
                    При выборе нового статуса и нажатии кнокпи <<запросить смену статуса>>
                    в очерез на отправку в \gloss{UNF} \index{1С УНФ} встаёт \gloss{zni} \index{ЗНИ}.
                    Об этом свидетельссвует информационное сообщение вверху страницы (см. рис.\ref{fig:admin_func_order_history}).
